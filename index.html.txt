// ===== CORE SYSTEM =====
const Core = {
    data: { services: [], offers: [], products: [], branches: [], gallery: [], settings: { salonName: 'صالوني', phone: '0560000000', address: 'الرياض' } },
    init() { this.load(); return this; },
    load() { try { const d = localStorage.getItem('SALON_DB'); if (d) this.data = JSON.parse(d); } catch(e){} },
    save() { try { localStorage.setItem('SALON_DB', JSON.stringify(this.data)); } catch(e){} },
    get(key) { return this.data[key] || []; },
    add(key, item) { if (!this.data[key]) this.data[key] = []; const id = Date.now()+Math.random().toString(36).substr(2,9); const newItem = {...item, id, createdAt:new Date().toISOString()}; this.data[key].push(newItem); this.save(); return newItem; },
    deleteByIndex(key, index) { if (!this.data[key] || !this.data[key][index]) return false; this.data[key].splice(index,1); this.save(); return true; },
    getSettings() { return this.data.settings; },
    updateSettings(settings) { this.data.settings = {...this.data.settings, ...settings}; this.save(); },
    export() { return JSON.stringify(this.data, null, 2); },
    import(jsonData) { try { this.data = JSON.parse(jsonData); this.save(); return true; } catch(e){ return false; } },
    getStats() { return { services: this.data.services.length, offers: this.data.offers.length, gallery: this.data.gallery.length, branches: this.data.branches.length, total: this.data.services.length + this.data.offers.length + this.data.gallery.length + this.data.branches.length }; }
};
Core.init();
window.Core = Core;

// ===== UPLOAD SYSTEM =====
const Upload = {
    CLOUD_NAME: 'YOUR_CLOUD_NAME',  // ⚠️ غير هذا
    UPLOAD_PRESET: 'YOUR_UPLOAD_PRESET', // ⚠️ وغير هذا
    async image(file) {
        if (!file) throw new Error('لم يتم تحديد ملف');
        const formData = new FormData();
        formData.append('file', file);
        formData.append('upload_preset', this.UPLOAD_PRESET);
        const res = await fetch(`https://api.cloudinary.com/v1_1/${this.CLOUD_NAME}/image/upload`, { method: 'POST', body: formData });
        if (!res.ok) throw new Error('فشل رفع الصورة');
        const data = await res.json();
        return data.secure_url;
    }
};
window.Upload = Upload;

// ===== APP =====
function updateSalonName() {
    const s = Core.getSettings();
    const name = s.salonName || 'صالوني';
    document.getElementById('salonName').textContent = name;
    document.getElementById('heroSalonName').textContent = name;
    document.getElementById('footerSalonName').textContent = name;
    document.getElementById('footerSalonName2').textContent = name;
    document.getElementById('footerPhone').textContent = s.phone || '0560000000';
    document.getElementById('footerAddress').textContent = s.address || 'نقدم أفضل خدمات التجميل';
    document.title = name;
}

function renderServices() {
    const c = document.getElementById('servicesContainer');
    const items = Core.get('services');
    if (!items.length) { c.innerHTML = '<div class="loading">لا توجد خدمات</div>'; return; }
    c.innerHTML = items.map(s => `
        <div class="item">
            ${s.image ? `<img src="${s.image}">` : `<div style="height:220px;background:linear-gradient(135deg,#667eea,#764ba2);display:flex;align-items:center;justify-content:center;color:#fff;font-size:48px"><i class="fas fa-cut"></i></div>`}
            <div class="item-content"><h3>${s.title||'خدمة'}</h3><p>${s.description||''}</p>${s.price ? `<span class="price">${s.price} ﷼</span>` : ''}</div>
        </div>
    `).join('');
}
